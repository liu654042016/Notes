        return root;

void preorder(TreeNode* root){
    visit(root);
    preorder(root->left);
    preorder(root->right);
}
void inorder(TreeNode* root){
    inorder(root->left);
    visit(root);
    inorder(root->right);
}
void postorder(TreeNode* root){
    postorder(root->left);
    postorder(root->right);
